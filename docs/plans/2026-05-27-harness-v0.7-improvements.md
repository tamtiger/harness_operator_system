> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Address 5 real-world pain points discovered during harness workflow testing on paymenthub-tenant-notifier-service.

**Architecture:** Incremental improvements to existing tools — no new MCP tools added, only enhanced behavior within `verify_run`, `scope_check`, and `session_handoff`. Plus a new structured response format for verify results.

**Tech Stack:** TypeScript, better-sqlite3, picomatch, Node.js 20+

---

## Pain Points Addressed

| # | Problem | Solution |
|---|---------|----------|
| 1 | `verify_run` returns `passed: false` for pre-existing lint issues unrelated to new code | Add `--changed-only` mode that filters lint to git-changed files only |
| 2 | Verify output truncated at 8KB, hard to know which step failed | Return structured `step_results` array with per-step pass/fail |
| 3 | Scope enforcement is advisory-only (agent can ignore) | N/A — keep advisory (enforcement is IDE-hook territory, not MCP tool) |
| 4 | No evidence auto-save when verify passes | Auto-save verify output to `.harness/evidence/{task_id}/` |
| 5 | `session_handoff` doesn't capture verify status | Add optional `verify_status` field to handoff data |

## Additional Improvements (from code review)

| # | Problem | Solution |
|---|---------|----------|
| 6 | `verify_run` stops on first failure — can't see all issues at once | Add `fail_fast` option (default true, set false to run all steps) |
| 7 | No way to know session duration | Track `duration_seconds` in session_end/handoff |
| 8 | `progress_log` entries lack structured metadata | Add optional `files_changed` field to progress entries |

```typescript
// src/lib/git-diff.ts
import { execSync } from "node:child_process";

/**
 * Parse `git diff --name-only` output into a list of relative file paths.
 */
export function parseGitDiffOutput(output: string): string[] {
  return output
    .split(/\r?\n/)
    .map((line) => line.trim())
    .filter((line) => line.length > 0);
}

/**
 * Get list of files changed relative to a base ref (default: HEAD).
 * Includes both staged and unstaged changes.
 * Returns empty array if not a git repo or git fails.
 */
export function getChangedFiles(
  repoPath: string,
  baseRef: string = "HEAD"
): string[] {
  try {
    // Get staged + unstaged changes vs base
    const output = execSync(
      `git diff --name-only ${baseRef} && git diff --name-only --cached`,
      {
        cwd: repoPath,
        encoding: "utf-8",
        timeout: 10_000,
        stdio: ["pipe", "pipe", "pipe"],
      }
    );
    // Deduplicate
    const files = parseGitDiffOutput(output);
    return [...new Set(files)];
  } catch {
    return [];
  }
}
```

```typescript
// src/lib/evidence.ts
import { existsSync, mkdirSync, readFileSync, writeFileSync } from "node:fs";
import { join } from "node:path";
import { resolveHarnessDir } from "./repo.js";

export interface EvidenceData {
  [key: string]: unknown;
  passed: boolean;
  saved_at?: string;
}

export interface SaveResult {
  saved: boolean;
  path: string;
}

/**
 * Save verify evidence for a task.
 * Stored at: .harness/evidence/{taskId}/verify.json
 */
export function saveEvidence(
  repoPath: string,
  taskId: string,
  data: Record<string, unknown>
): SaveResult {
  const harnessDir = resolveHarnessDir(repoPath);
  const evidenceDir = join(harnessDir, "evidence", taskId);

  if (!existsSync(evidenceDir)) {
    mkdirSync(evidenceDir, { recursive: true });
  }

  const filePath = join(evidenceDir, "verify.json");
  const payload = {
    ...data,
    saved_at: new Date().toISOString(),
  };

  writeFileSync(filePath, JSON.stringify(payload, null, 2), "utf-8");
  return { saved: true, path: filePath };
}

/**
 * Read saved evidence for a task. Returns null if not found.
 */
export function readEvidence(
  repoPath: string,
  taskId: string
): EvidenceData | null {
  const harnessDir = resolveHarnessDir(repoPath);
  const filePath = join(harnessDir, "evidence", taskId, "verify.json");

  if (!existsSync(filePath)) return null;

  try {
    return JSON.parse(readFileSync(filePath, "utf-8"));
  } catch {
    return null;
  }
}
```

- [ ] **Step 8: Add `buildFilteredLintCmd` helper**

```typescript
function buildFilteredLintCmd(
  originalCmd: string,
  changedFiles: string[],
  runtime: { runtime: string }
): string {
  // Filter to only lintable files
  const lintableExts: Record<string, string[]> = {
    node: [".ts", ".tsx", ".js", ".jsx"],
    dotnet: [".cs"],
    python: [".py"],
    go: [".go"],
    rust: [".rs"],
  };

  const exts = lintableExts[runtime.runtime] || [];
  const filtered = changedFiles.filter((f) =>
    exts.some((ext) => f.endsWith(ext))
  );

  if (filtered.length === 0) return "echo No lintable changed files";

  // Build command based on runtime
  if (runtime.runtime === "dotnet") {
    const includes = filtered.map((f) => `--include "${f}"`).join(" ");
    return `dotnet format --verify-no-changes ${includes}`;
  }

  if (runtime.runtime === "node") {
    // Assume eslint or similar — append file list
    const fileList = filtered.map((f) => `"${f}"`).join(" ");
    return `${originalCmd} ${fileList}`;
  }

  // Fallback: run original command
  return originalCmd;
}
```

```typescript
export function sessionEnd(sessionId: string): SessionEndResult {
  const db = getDb();
  const now = new Date().toISOString();

  // Get started_at for duration calc
  const session = db
    .prepare(`SELECT started_at FROM sessions WHERE id = ?`)
    .get(sessionId) as { started_at: string } | undefined;

  if (!session) {
    throw new Error(`Session not found: ${sessionId}`);
  }

  const durationSeconds = Math.round(
    (Date.now() - new Date(session.started_at).getTime()) / 1000
  );

  db.prepare(`UPDATE sessions SET status = 'closed', ended_at = ? WHERE id = ?`)
    .run(now, sessionId);

  return { session_id: sessionId, status: "closed", duration_seconds: durationSeconds };
}
```

```typescript
export function handoffWrite(
  repoPath: string,
```
